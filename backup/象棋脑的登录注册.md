(function(Scratch) {
    'use strict';

    class LoginExtension {
        constructor() {
            this.users = {}; 
            this.ws = null;
            this.isOnline = false; 
            this.connectToServer();
        }

        connectToServer() {
            try {
                this.ws = new WebSocket('ws://localhost:8080'); 
                this.ws.onopen = () => {
                    this.isOnline = true;
                    console.log('Connected to cloud server');
                };
                this.ws.onmessage = (event) => {
                    const data = JSON.parse(event.data);
                    if (data.type === 'sync') {
                        this.users = data.users;
                    }
                };
                this.ws.onclose = () => {
                    this.isOnline = false;
                    console.log('Disconnected from cloud server');
                };
                this.ws.onerror = () => {
                    this.isOnline = false;
                };
            } catch (e) {
                this.isOnline = false;
            }
        }

        getInfo() {
            return {
                id: 'login',
                name: '登录扩展',
                blocks: [
                    {
                        opcode: 'register',
                        blockType: Scratch.BlockType.COMMAND,
                        text: '注册 用户名 [username] 密码 [password]',
                        arguments: {
                            username: {
                                type: Scratch.ArgumentType.STRING,
                                defaultValue: 'user'
                            },
                            password: {
                                type: Scratch.ArgumentType.STRING,
                                defaultValue: 'pass'
                            }
                        }
                    },
                    {
                        opcode: 'login',
                        blockType: Scratch.BlockType.BOOLEAN,
                        text: '登录 用户名 [username] 密码 [password]',
                        arguments: {
                            username: {
                                type: Scratch.ArgumentType.STRING,
                                defaultValue: 'user'
                            },
                            password: {
                                type: Scratch.ArgumentType.STRING,
                                defaultValue: 'pass'
                            }
                        }
                    },
                    {
                        opcode: 'isOnline',
                        blockType: Scratch.BlockType.BOOLEAN,
                        text: '是否已联网'
                    }
                ]
            };
        }

        register(args) {
            const { username, password } = args;
            if (this.users[username]) {
                return; // 用户已存在
            }
            this.users[username] = password;
            if (this.ws && this.ws.readyState === WebSocket.OPEN) {
                this.ws.send(JSON.stringify({ type: 'sync', users: this.users })); // 同步到云端
            }
        }

        login(args) {
            const { username, password } = args;
            return this.users[username] === password;
        }

        isOnline() {
            return this.isOnline;
        }
    }

    Scratch.extensions.register(new LoginExtension());
})(Scratch);
